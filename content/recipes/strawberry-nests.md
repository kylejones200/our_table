ingredients:

  - "1/2 cup sugar"
  - "2 tsp milk"
  - "1 tsp vanilla"
  - "2 pkg. shortcake dessert cups"
  - "3 1/2 oz. flaked coconut"
  - "1 quart fresh strawberrie, sliced with sugar"

steps:

  - "Spoon sliced strawberries over the coated dessert cups."
